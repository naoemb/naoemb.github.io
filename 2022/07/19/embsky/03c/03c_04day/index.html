/*
“百钱百鸡”问题,百钱买百鸡,公鸡1个3块钱,母鸡1个2块钱，小鸡3个一块钱,问公鸡,母鸡,小鸡各多少个
*/
#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[])
{
int g = 0, m = 0, x = 0;//分别表示公鸡 母鸡 小鸡的个数

for(g = 0; g < 33; g++)
{
    for(m = 0; m < 50; m++)
    {
        for(x = 0; x < 100; x += 3)
        {
            if((g*3+m*2+x/3==100) && (g+m+x==100))
                printf("G:%d M:%d X:%d\n", g, m, x);
        }
    }
}

return 0;

}

/*
随机函数
rand(); //功能:就是用来产生随机数的
rand() % 1000; //[0 ~ 999]
rand() % 100 - 50;//[-50 ~ 49]

srand();        //功能:可以保证每一次产生的随机数可变
    srand函数在调用时需要有可变的因子
    time(NULL);    //时间种子
    getpid();    //进程种子

*/
#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <sys/types.h>
#include <unistd.h>

int main(int argc, char *argv[])
{
int sum = 0;
int i = 0;

//srand(time(NULL));
//注意:srand函数在程序中只能调用一次
srand(getpid());

for(i = 0; i < 5; i++)
{
    sum = rand() % 100 - 50;
    printf("sum = %d\n", sum);
}

return 0;

}

/*

#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <sys/types.h>
#include <unistd.h>

int main(int argc, char *argv[])
{
int T = 0;//存储随机出的数字
int guess = 0;//存储玩家猜的数字
int num = 1;

srand(getpid());//随机种子

T = rand() % 10;

for(num = 1; ; num++)
{
    scanf("%d", &guess);
    if(T > guess)
        printf("[%d]    Small!\n", num);
    else if(T < guess)
        printf("[%d]    Big!\n", num);
    else
    {
        printf("[%d]    Bingo!\n", num);
        break;
    }
}

return 0;

}
