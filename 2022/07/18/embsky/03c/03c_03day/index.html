/*
输入年月日,打印出是该年的第几天
//2022-7-18
是2022年的第?天
2022-1-9 9天
2022-2-9 9+31天
2022-3-9 9+(28||29)+31
2022-4-9 9+31+(28||29)+31
*/
#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[])
{
int year = 0, month = 0, day = 0;
int flag = 0;//用来标记闰年和平年
int days = 0;//用来存储最后总天数

scanf("%d-%d-%d", &year, &month, &day);
/*if error*/

flag = (!(year%4)&&(year%100)) || !(year%400);
//如果是闰年 flag = 1 否则 flag = 0

switch(month)
{
    case 1 : days += day; break;
    case 2 : days += (day + 31); break;
    case 3 : days += (day + 59 + flag); break;
    case 4 : days += (day + 90 + flag); break;
    case 5 : days += (day + 120 + flag); break;
    case 6 : days += (day + 151 + flag); break;
    case 7 : days += (day + 181 + flag); break;
    case 8 : days += (day + 212 + flag); break;
    case 9 : days += (day + 243 + flag); break;
    case 10: days += (day + 273 + flag); break;
    case 11: days += (day + 304 + flag); break;
    case 12: days += (day + 334 + flag); break;
}

printf("%d年%d月%d日是这一年的第%d天!\n", year, month, day, days);

return 0;

}

/*
输入年月日,打印出是该年的第几天
//2022-7-18

2022-1-9 9天
2022-2-9 9+31天
2022-3-9 9+(28||29)+31
2022-4-9 9+31+(28||29)+31
*/
#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[])
{
int year = 0, month = 0, day = 0;
int flag = 0;//用来标记闰年和平年
int days = 0;//用来存储最后总天数

scanf("%d-%d-%d", &year, &month, &day);
/*if error*/

flag = (!(year%4)&&(year%100)) || !(year%400);
//如果是闰年 flag = 1 否则 flag = 0

switch(month)
{
    case 12: days += 30;
    case 11: days += 31;
    case 10: days += 30;
    case 9 : days += 31;
    case 8 : days += 31;
    case 7 : days += 30;
    case 6 : days += 31;
    case 5 : days += 30;
    case 4 : days += 31;
    case 3 : days += (28 + flag);
    case 2 : days += 31;
    case 1 : days += day;
}

printf("%d年%d月%d日是这一年的第%d天!\n", year, month, day, days);

return 0;

}

/*
打印0~100以内所有的质数(素数)
(只能被1 和 本身整除的数)
*/
#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[])
{
int i = 0, j = 0;
int num = 1;

for(i = 2; i < 100; i++)
//该循环为了取出[0-100]以内所有数字
{
    for(j = 2; j < i; j++)//该循环为了取出[2 - i-1]
    {
        if(i % j == 0)//判断是否能被整除
        {
            break;//如果能被整除说明不是质数,跳出
        }
    }
    if(i == j)
        printf("[%d]    %d\n", num++, i);
}

return 0;

}
