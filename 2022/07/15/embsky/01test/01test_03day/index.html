数据类型 变量名;
1)基本的数据类型
整型 有符号/无符号 short int long

3)数据类型的大小以及范围在头文件中有记录
/usr/include/limits.h

*/

{//char short int long float double

printf(“sizeof(long) = %ld\n”, sizeof(long));
printf(“sizeof(float) = %ld\n”, sizeof(float));

{

printf("a >= b = %d\n", a >= b);
printf("a <= b = %d\n", a <= b);
printf("a != b = %d\n", a != b);

return 0;
