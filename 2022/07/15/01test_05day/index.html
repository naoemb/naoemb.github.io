/*
三角形
*/
#include <stdio.h>

int main(void)
{
int a = 0, b = 0, c = 0;//使用a b c变量表示三角形的三个边

scanf("%d-%d-%d", &a, &b, &c);//通过scanf函数录入三个边
/*if error*/

if((a+b>c) && (a+c>b) && (b+c>a))//罗列普通三角形的条件
{
    if((a*a+b*b==c*c) || 
    (a*a+c*c==b*b) || 
    (b*b+c*c==a*a))//罗列直角三角形的条件
    {
        printf("可以构成直角三角形!\n");
    }
    else if(a==b || a==c || b==c)//罗列等腰三角形的条件
    {
        if(a==b && a==c)//罗列等边三角形的条件
        {
            printf("可以构成等边三角形!\n");
        }
        else
        {
            printf("可以构成等腰三角形!\n");
        }
    }
    else
    {
        printf("可以构成普通三角形!\n");
    }
}
else
{
    printf("不能构成三角形!\n");
}

return 0;

}

/*
从小到大打印
*/
#include <stdio.h>

int main(void)
{
int x = 0, y = 0, z = 0;
int tmp = 0;

scanf("%d-%d-%d", &x, &y, &z);
//x获取三个值中的最小值 z获取三个值中的最大值

if(x > y)
{
    tmp = x;
    x = y;
    y = tmp;
}

if(x > z)
{
    tmp = x;
    x = z;
    z = tmp;
}

if(y > z)
{
    tmp = y;
    y = z;
    z = tmp;
}

printf("%d<%d<%d\n", x, y, z);

return 0;

}
